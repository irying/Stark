> 209 给定一个整型数组和一个数字s，找到数组中最短的一个连续子数组，使得连续子数组的数字和sum>=s，返回这个最短的连续子数组的返回值。比如0到6数组中，3和4是最短的一个连续子数组。

```
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

class Solution {
public:
    vector<int> minSubArrayLen(int s, vector<int>&nums) {
        int l = 0, r = -1;
        int sum = 0;
        int res = nums.size() + 1;
        while(l < nums.size()) {
            if (r+1 < nums.size() && sum<s)
                sum += nums[++r];
            else
                sum -= nums[l++];
            
            if(sum >=s )
                res = min(res, r-l+1);
        }
        vector<int> result;
        if(res == nums.size() + 1) {
            result.push_back(0);
            result.push_back(0);
            return result;
        }
           
        
        result.push_back(l);
        result.push_back(r);
            
        return result;
        // return res;
    }
};
int main(){
    Solution solution;
    vector<int> num;
    num.push_back(1);
    num.push_back(2);
    num.push_back(2);
    num.push_back(3);
    num.push_back(4);
    vector<int> res = solution.minSubArrayLen(7, num);
    for(int i = 0;i < res.size();i++){
       
        cout<<res[i]<<endl;
    }
    return 0;
}
```

## 3.在一个字符串中寻找没有重复字母的最长字串

```C
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

class Solution {
public:
    int minSubArrayLen(string s) {
        int freq[256] = {0};
        int l = 0, r = -1;
        int res = 0;
       
        while(l < s.size()) {
            if (r+1 < s.size() && freq[s[r+1]] == 0)
                freq[s[++r]] ++;
            else
                freq[s[l++]] --;
            
            res = max(res, r-l+1);
        }
       
        return res;
    }
};
int main(){
    Solution solution;
   
    int res = solution.minSubArrayLen("abcdacdb");

       
        cout<<res<<endl;
    return 0;
}
```
