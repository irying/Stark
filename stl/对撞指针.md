```
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int> &numbers, int target) {
        int i=0,mid=0;
        int length = numbers.size();
        for(i;i<length;i++) {
            int num2 = target-numbers[i];
            int l = i+1,r = length-1;
            while(l<=r) {
                mid = l + (r-l)/2;
                if(numbers[mid] == num2) {
                    vector<int> result;
                    result.push_back(i+1);
                    result.push_back(mid+1);
                    return result;
                }
                
                if(numbers[mid] < num2)
                    l = mid + 1;
                else
                    r = mid - 1;
            }
        }
    }
};

int main(){
    Solution solution;
    vector<int> num;
    num.push_back(1);
    num.push_back(2);
    num.push_back(3);
    num.push_back(4);

    vector<int> res = solution.twoSum(num, 5);
    // 输出
    for(int i = 0;i < res.size();i++){
       
        cout<<res[i]<<endl;
    }
    return 0;
}
```
